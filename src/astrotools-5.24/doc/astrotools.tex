% To avoid awkward vertical spacing between text while in \flushbottom mode,
% \raggedbottom is used.  But, footnotes will not always appear at the very
% bottom of the page.  To fix this, \skip\footins is redefined to do some
% filling.  In turn, this causes a problem where \section or other commands
% may go onto the next page prematurely.  This occurs because \skip\footins'
% filling took away space from the page goal, even though the filling is a
% rubber length.  To counteract this problem, \count\footins was changed to
% 250 (from 1000).  Basically, the output routine thinks that footnotes take
% up less space than they really do. This appears to work in most situations.
%
\raggedbottom
\skip\footins=  0.10in plus 1fil\relax	% Force footnotes to be \flushbottom
\count\footins=  250			% Undo loss of regular text to next page

%- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
%
% \begin{indpar}{leftindent}{rightindent}%
% \end{indpar}
%
% Provide a paragraphing environment where the left and/or right margin are
% indented.  Both values must be positive length specifications (thus enter
% 0pt, not just 0).  The left margin is indented by LEFTINDENT units and the
% right margin is indented by RIGHTINDENT units.
%
% NOTE: The comment character (%) should be placed immediately after the right
%       indentation specification.  This prevents a small amount of indentation
%       from being made on the first line within the indpar environment.
%
\newlength{\indparwidth}		% Define globally
\newlength{\indparindent}		% Define globally
\newlength{\indparskip}			% Define globally
%
\newenvironment{indpar}[2]%
{%
\setlength{\indparwidth}{\textwidth}%
\advance\indparwidth by -#1\relax%
\advance\indparwidth by -#2\relax%
\setlength{\indparindent}{\parindent}%
\setlength{\indparskip}{\parskip}%
\hspace*{#1}\begin{minipage}{\indparwidth}%
\setlength{\parindent}{\indparindent}%
\setlength{\parskip}{\indparskip}%
\par\noindent%
}%
{%
\end{minipage}%
}

% \rightfacing
%
% Determine whether we're on an odd page.  If so, generate a blank page or at
% least bump up the page number.  This macro is useful at the beginning of
% chapters (before the \chapter command), etc. to force the page to be right
% facing.
%
\newcommand{\rightfacing}%
{%
\clearpage%
\ifodd\count0\else%
   \ifodd\userblankpages%
      \thispagestyle{empty}\mbox{}\clearpage%
   \else%
      \addtocounter{page}{1}%
   \fi%
\fi%
}

%
% \epsfigure{file}{title}{ref-label}
%
% Include an Encapsulated PostScript file containing a figure into the document.
% The FILE specification, a TITLE, and a REFerence LABEL must be provided.  The
% reference label can be used within the text as such:
%
%   Figure \ref{REF-LABEL} shows ...
%
% NOTE:	There was a problem in using the figure environment when \raggedbottom
%	is used along with the \skip\footins and \count\footins redefinitions.
%	Captions are placed in a figure environment that must go HERE!  But,
%	when the figure is near the bottom of the page, and there are footnotes
%	present, it determines that there really is not enough space. Thus, the
%	caption gets to go onto its own page (the page of floats stuff)!
%
\def\epsfigure#1#2#3%
{%
\epsfxsize=0.98\textwidth%
\vskip 0.25in%
\goodbreak%
{%
\samepage%
\epsfbox{#1}%
\vskip -1.6ex%			% Prevent \caption ending up on a page of floats
\nopagebreak%
\begin{figure}[h]
\caption{#2}%
\label{#3}%
\end{figure}%
}%
}
